<script>

const socket=io();

let applied=0,skipped=0,errors=0,total=0;


fetch('/ai_mode').then(r=>r.json()).then(d=>{

  const tog=document.getElementById('aiToggle');

  const txt=document.getElementById('aiModeText');

  if(d.mode==='premium'){tog.checked=true;txt.textContent='Premium';txt.style.color='#a78bfa';}

  else{txt.textContent='Free';txt.style.color='#34d399';}

});


function toggleAI(el){

  const mode=el.checked?'premium':'free';

  const txt=document.getElementById('aiModeText');

  fetch('/ai_mode',{method:'POST',headers:{'Content-Type':'application/json'},body:JSON.stringify({mode})})

    .then(r=>r.json()).then(()=>{

      txt.textContent=mode==='premium'?'Premium':'Free';

      txt.style.color=mode==='premium'?'#a78bfa':'#34d399';

      addLog(`⚡ AI mode → ${mode.toUpperCase()}`,'i');

    });

}


function toggleTheme(){

  const html=document.documentElement;

  const dark=html.getAttribute('data-theme')==='dark';

  html.setAttribute('data-theme',dark?'light':'dark');

  document.getElementById('themeIco').textContent=dark?'🌙':'☀️';

}


function setRunning(on,platform=''){

  const dot=document.getElementById('sdot');

  const txt=document.getElementById('stxt');

  const btns=['btnLI','btnDI','btnBoth'];

  if(on){

    dot.className='sdot running';

    txt.textContent=`RUNNING ${platform.toUpperCase()}`;

    btns.forEach(b=>document.getElementById(b).disabled=true);

    document.getElementById('btnStop').disabled=false;

  } else {

    dot.className='sdot';

    txt.textContent='IDLE';

    btns.forEach(b=>document.getElementById(b).disabled=false);

    document.getElementById('btnStop').disabled=true;

  }

}


function runBot(platform){

  applied=0;skipped=0;errors=0;

  updateStats();

  setRunning(true,platform);

  addLog(`🚀 Starting ${platform} bot...`,'i');

  fetch(`/run/${platform}`,{method:'POST'}).then(r=>r.json()).then(d=>{

    if(d.status==='already_running'){addLog('⚠️ Bot already running!','e');setRunning(false);}

  });

}


function stopBot(){

  fetch('/stop',{method:'POST'});

  addLog('⏹ Stop requested...','e');

  setRunning(false);

}


function updateStats(){

  document.getElementById('sApplied').textContent=applied;

  document.getElementById('sSkipped').textContent=skipped;

  document.getElementById('sErrors').textContent=errors;

  document.getElementById('sTotal').textContent=total;

}


function addLog(msg,type=''){

  const box=document.getElementById('logBox');

  const div=document.createElement('div');

  let cls='ll';

  if(msg.includes('✅')||msg.includes('Applied!')||msg.includes('Logged'))cls='ll s';

  else if(msg.includes('❌')||msg.includes('Error'))cls='ll e';

  else if(msg.includes('⏭️')||msg.includes('🚫')||msg.includes('skipping')||msg.includes('Not in'))cls='ll k';

  else if(msg.includes('💼')||msg.includes('Applying'))cls='ll a';

  else if(type==='i'||msg.includes('🔍')||msg.includes('🚀')||msg.includes('Starting'))cls='ll i';

  div.className=cls;

  const t=new Date().toLocaleTimeString('en',{hour:'2-digit',minute:'2-digit',second:'2-digit',hour12:false});

  div.textContent=`[${t}] ${msg}`;

  box.appendChild(div);

  box.scrollTop=box.scrollHeight;

  if(msg.includes('Applied!')){applied++;total++;}

  if(msg.includes('skipping')||msg.includes('Not in')||msg.includes('🚫'))skipped++;

  if(msg.includes('❌ Error'))errors++;

  updateStats();

}


function clearLogs(){document.getElementById('logBox').innerHTML='<div class="ll i">// logs cleared</div>';}


socket.on('log',d=>addLog(d.msg));

socket.on('done',d=>{setRunning(false);addLog(`🎉 Done! Applied: ${d.stats.applied}`,'s');loadTracker();});


function loadTracker(){

  fetch('/tracker').then(r=>r.json()).then(d=>{

    const head=document.getElementById('tHead');

    const body=document.getElementById('tBody');

    if(d.error){body.innerHTML=`<tr><td colspan="6" class="empty-state">${d.error}</td></tr>`;return;}

    if(!d.rows||!d.rows.length){body.innerHTML='<tr><td colspan="6" class="empty-state">No applications yet</td></tr>';return;}

    head.innerHTML='<tr>'+d.cols.map(c=>`<th>${c}</th>`).join('')+'</tr>';

    body.innerHTML=d.rows.slice().reverse().map(row=>

      '<tr>'+row.map((cell,i)=>{

        if(d.cols[i]==='Status'){const cls=cell==='Applied'?'pill-applied':cell==='Interview'?'pill-interview':'pill-rejected';return `<td><span class="pill ${cls}">${cell||'-'}</span></td>`;}

        if(d.cols[i]==='Link'&&cell)return `<td><a class="tlink" href="${cell}" target="_blank">↗ view</a></td>`;

        return `<td title="${cell}">${cell||'-'}</td>`;

      }).join('')+'</tr>'

    ).join('');

  });

}

loadTracker();
